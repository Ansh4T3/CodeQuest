4. Factorial of a Number

#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    long long fact = 1;
    for (int i = 1; i <= n; i++) {
        fact *= i;
    }

    printf("%lld", fact);
    return 0;
}

5. Reverse a Number

#include <stdio.h>

int main() {
    long long n, rev = 0;
    scanf("%lld", &n);

    while (n > 0) {
        rev = rev * 10 + n % 10;
        n /= 10;
    }

    printf("%lld", rev);
    return 0;
}

6. Palindrome Number

#include <stdio.h>

int main() {
    long long n, original, rev = 0;
    scanf("%lld", &n);

    original = n;
    while (n > 0) {
        rev = rev * 10 + n % 10;
        n /= 10;
    }

    if (rev == original)
        printf("Palindrome");
    else
        printf("Not Palindrome");

    return 0;
}


7.Prime number check

#include <stdio.h>
#include <math.h>

int main() {
    int n;
    if (scanf("%d", &n) != 1) {
        return 1; // error reading input
    }

    if (n <= 1) {
        printf("No");
        return 0;
    }

    int isPrime = 1;
    for (int i = 2; i <= sqrt(n); i++) {
        if (n % i == 0) {
            isPrime = 0;
            break;
        }
    }

    if (isPrime) {
        printf("Yes");
    } else {
        printf("No");
    }

    return 0;
}
